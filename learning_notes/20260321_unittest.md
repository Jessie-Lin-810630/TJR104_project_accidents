# command for testing
    ```
    # 運行所有測試
        poetry run pytest test/unit_test/ -v

        # 運行特定檔案
        poetry run pytest test/unit_test/test_util_create_engine_to_mysql.py -v

        # 運行特定測試類
        poetry run pytest test/unit_test/test_util_create_engine_to_mysql.py::TestCreateEngineToMysql -v

        # 查看覆蓋率
        poetry run pytest test/unit_test/ --cov=src --cov-report=term-missing
    ```
